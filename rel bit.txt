#include <stdio.h>
void sol(int n,int l)
{
  
 int i,k,g=0,g1=0,g2=0,r,x,an;
 for(i=1;i<=(n-1);i++)
 {
     for(k=n;k>i;k--)
     {
         r=i|k;
 an=i&k;
 x=i^k;
 if(r>g&&r<l)
 {
     g=r;
 }
 if(an>g1&&an<l)
 {
     g1=an;
 }
 if(x>g2&&x<l)
 {
     g2=x;
 }
     
     }
 }
 
 printf("%d\n%d\n%d",g1,g,g2);
}
int main()
{ int n,l;
 
 scanf("%d%d",&n,&l);
 sol(n,l);
 

    return 0;
}
